	\begin{enumerate}
    
    \item [а)] 
	    $F_n = 
            \begin{cases}
            0 &\text{при $n = 0;$}\\
            1 &\text{при $n = 1;$}\\
            F_{n-1} + 2 F_{n-2} &\text{при $n \geq 2.$}
        \end{cases}$\\
        
        $f(x) = F_0 + F_1x + F_2x^2 + F_3x^3 + ... + F_nx^n + ... + $\\
        
        $2x \cdot f(x) = 2F_0x + 2F_1x^2 + 2F_2x^3 + 2F_3x^4 + ... + 2F_nx^{n+1} + ... + $\\
        
        $f(x) + 2x \cdot f(x) = F_0 + (F_1 + 2F_0)x + (F_2 + 2F_1)x^2 + (F_3 + 2F_2)x^3 + ... + (F_n + 2F_{n-1})x^n + ... +$\\
        
        (1 + 2x) $f(x)  = F_2x + F_3x^2 + F_4x^3 + ... + F_{n+1}x^n + ... +$ ($F_{n+1} = F_n + 2F_{n-1}$)\\
        
        $x(1 + 2x)f(x) + x + 0 = F_0 + F_1x + F_2x^2 + F_3x^3 + ... + F_nx^n + ... +$\\
        
        $$x(f(x)(1 + 2x) + 1) = f(x) \Rightarrow f(x) = \dfrac{x}{(1 - 2x)(1 + x)}$$\\
    
    \item [б)]
        $F_n = 
        \begin{cases}
           1 &\text{при $n = 0;$}\\
           3 &\text{при $n = 1;$}\\
           4 F_{n-1} - 4 F_{n-2} &\text{при $n \geq 2.$}
        \end{cases}$\\
         
        $f(x) = F_0 + F_1x + F_2x^2 + F_3x^3 + ... + F_nx^n + ... +$\\
        
        $4f(x) =  4F_0 + 4F_1x + 4F_2x^2 + 4F_3x^3 + ... + 4F_nx^n + ... +$\\
        
        $4f(x) \cdot x = 4F_0x + 4F_1x^2 + 4F_2x^3 + 4F_3x^4 + ... + 4F_nx^{n+1} + ... +$\\
        
        $4f(x) - 4f(x) \cdot x =  4F_0 + (4F_1 - 4F_0)x + (4F_2 - 4F_1)x^2 + (4F_3 - 4F_2)x^3 + ... + (4F_n - 4F_{n-1})x^n + ... +$\\
        
        $4f(x)(1 - x) = 4 + F_2x + F_3x^2 + F_4x^3 + ... + F_{n+1}x^n + ... +$ ($F_{n+1} = 4F_n - 4F_{n-1})$\\
        
        $1 + x \cdot (4f(x)(1 - x) - 1) = F_0 + F_1x + F_2x^2 + F_3x^3 + ... + F_nx^n + ... +$\\
        
        $$1 + x(4f(x)(1 - x) - 1) = f(x) \Rightarrow f(x) = \dfrac{1 - x}{(1 - 2x)^2}$$\\
          
	\end{enumerate}
