	\begin{enumerate}
	    \item [а)] 
	    Разложим в ряд Тейлора следующую функцию:
	    \[
	        (1 - x)^{-1} = 1 + x + x^2 + x^3 \dots 
	    \]
	    \[
	        \Rightarrow ((1 - x)^{-1})' \cdot x = x + 2 \cdot x^2 + 3 \cdot x^3 \dots = \dfrac{x}{(1 - x)^2}
	    \]
	    
	    \item [б)]
	    Аналогично в ряд Тейлора:
	    \[
	        e^x = 1 + \dfrac{x}{1!} + \dfrac{x^2}{2!} + \dfrac{x^3}{3!}
	    \]
	    
	    Это и есть, что нам нужно.
	\end{enumerate}
	
    %4
	\textbf{№4}
	\\
	Доказать, что:
	\[
	    \sum\limits_{k = 2}^{n} k(k-1)C_n^k = n(n-1)2^{n-2}
	\]
    
    Мы знаем, что: 
    \[
        (1 + x)^n = \sum\limits_{k = 0}^{n} C_n^k \cdot x^k = f(x)
    \]

    \[
        \Rightarrow f''(x) = n(n - 1)(1 + x)^{n - 2}
    \]
    
    С другой стороны:
    
    \[
        f''(x) = \sum\limits_{k = 2}^{n} k(k-1)C_n^k \cdot x^{k - 2}
    \]

    Получаем то, что и требовалось доказать:
    
    \[
        \sum\limits_{k = 2}^{n} k(k-1)C_n^k = f''(1) = n(n-1)2^{n-2}
    \]
